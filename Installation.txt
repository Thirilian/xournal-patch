# Dépendances de compilation
sudo apt-get update
sudo apt-get install curl cmake ninja-build libgtk-3-dev libpoppler-glib-dev portaudio19-dev libsndfile-dev \
  dvipng texlive libxml2-dev liblua5.4-dev libzip-dev librsvg2-dev gettext \
  libgtksourceview-4-dev help2man libqpdf-dev git

# cloner les sources
cd ~/Documents
git clone https://github.com/xournalpp/xournalpp
cd xournalpp

# Appliquer le patch
curl -O  https://raw.githubusercontent.com/Thirilian/xournal-patch/main/apply_paste_follow_cursor_v3.py
curl -O  https://raw.githubusercontent.com/Thirilian/xournal-patch/main/apply_txt_prefill.py
curl -O  https://raw.githubusercontent.com/Thirilian/xournal-patch/main/apply_resize_ratio_v3.py
curl -O  https://raw.githubusercontent.com/Thirilian/xournal-patch/main/apply_force_recompile.py
curl -O  https://raw.githubusercontent.com/Thirilian/xournal-patch/main/apply_color_widget_v3.py
curl -O  https://raw.githubusercontent.com/Thirilian/xournal-patch/main/apply_floating_marks_v2.py
curl -O  https://raw.githubusercontent.com/Thirilian/xournal-patch/main/apply_arrow_resize_fix_v2.py

python3 apply_paste_follow_cursor_v3.py
python3 apply_txt_prefill.py
python3 apply_resize_ratio_v3.py
python3 apply_force_recompile.py
python3 apply_color_widget_v3.py
python3 apply_floating_marks_v2.py
python3 apply_arrow_resize_fix_v2.py

# Compiler
mkdir build
cd build
cmake .. -DCMAKE_INSTALL_PREFIX=install -G Ninja -DCMAKE_BUILD_TYPE=RelWithDebInfo
cmake --build .
sudo cmake --build . --target install
sudo cmake --install . --prefix /usr/local # Replace xournalapp /usr/bin/xournalpp binary with the app you built
                                           # Can lead to unexpected behaviours. Do it at your own risks
                                           # In case of problems, reinstall xournalapp from your system package handler

# retirer les fichiers de build
cd
sudo rm -rf ~/Documents/xournalpp

xournalpp # launch xournal after build and installation
