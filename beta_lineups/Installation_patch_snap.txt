# Dépendances de compilation
sudo apt-get update
sudo apt-get install curl cmake ninja-build libgtk-3-dev libpoppler-glib-dev portaudio19-dev libsndfile-dev \
  dvipng texlive libxml2-dev liblua5.4-dev libzip-dev librsvg2-dev gettext \
  libgtksourceview-4-dev help2man libqpdf-dev git

# cloner les sources
cd ~/Documents
git clone https://github.com/xournalpp/xournalpp
cd xournalpp

# Appliquer le patch
curl -O  https://raw.githubusercontent.com/Thirilian/xournal-patch/main/beta_lineups/apply_alignment_snap_v1.py
curl -O  https://raw.githubusercontent.com/Thirilian/xournal-patch/main/beta_lineups/apply_alignment_snap_v2.py
curl -O  https://raw.githubusercontent.com/Thirilian/xournal-patch/main/beta_lineups/apply_alignment_snap_v3.py
curl -O  https://raw.githubusercontent.com/Thirilian/xournal-patch/main/beta_lineups/apply_alignment_snap_v4.py
python3 apply_alignment_snap_v1.py
python3 apply_alignment_snap_v2.py
python3 apply_alignment_snap_v3.py
python3 apply_alignment_snap_v4.py


# Compiler
mkdir build
cd build
sudo cmake .. -DCMAKE_INSTALL_PREFIX=install -G Ninja -DCMAKE_BUILD_TYPE=RelWithDebInfo
sudo cmake --build .
sudo cmake --build . --target install
sudo cmake --install . --prefix /usr/local
xournalpp

# retirer les fichiers de build
cd
sudo rm -rf ~/Documents/xournalpp
