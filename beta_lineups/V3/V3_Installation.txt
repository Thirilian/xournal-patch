# cloner les sources
cd ~/Documents
git clone https://github.com/xournalpp/xournalpp
cd xournalpp

curl -O  https://raw.githubusercontent.com/Thirilian/xournal-patch/main/apply_arrow_resize_fix_v2.py
python3 apply_arrow_resize_fix_v2.py

cp ~/Téléchargements/apply_alignment_snap_v90.py .
python3 apply_alignment_snap_v90.py

cp ~/Téléchargements/apply_alignment_snap_v11_1.py .
python3 apply_alignment_snap_v11_1.py
cp ~/Téléchargements/apply_alignment_snap_v11_2.py .
python3 apply_alignment_snap_v11_2.py

# Compiler
mkdir build
cd build
sudo cmake .. -DCMAKE_INSTALL_PREFIX=install -G Ninja -DCMAKE_BUILD_TYPE=RelWithDebInfo
sudo cmake --build .
sudo cmake --build . --target install
sudo cmake --install . --prefix /usr/local
xournalpp

# retirer les fichiers de build
cd
